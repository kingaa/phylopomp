269 void reassort (name_t i = 0, name_t j = 0, name_t* seg = NULL, name_t size = 1) {
270 std::string e = "reassort";
271 retimes.push_back(time());
272 bool exist;
273 if (size < 1 || seg == NULL) err("Empty segment array!");
274 if (size > nseg) err("Requested number of segments larger the total number.");
275 if (size == nseg) err("All segments reassort simultaneously!");
276
277 int inda, indb;
278 ball_t *a, *b;
279 inda = random_integer((inventory[0])[i].size());
280 indb = random_integer((inventory[0])[j].size());
281 while ((i==j) && (inda==indb)) indb = random_integer((inventory[0])[j].size());
282
283 for (name_t k = 0; k < nseg; k++) {
284 exist = anyof(seg, size, k);
285 if (exist) {
286 a = inventory[k].get_ball_idx(inda,i);
287 b = inventory[k].get_ball_idx(indb,i);
288 geneal[k].reassort(a,b,time());
289 } else {
290 geneal[k].update_uniq();
291 }
292 }
293 valid_invens(e);
294 };
296 void batch_sample (double frac = 0) {
297 size_t N = 0, deme_size[ndeme];
298 name_t i;
299 for (i = 0; i < ndeme; i++) {
300 deme_size[i] = (inventory[0])[i].size();
301 N += deme_size[i];
302 }
303 size_t n = (size_t)(N*frac);
304 if (n > 0) {
305 std::string e = "batch sample";
306 ball_t* a;
307 name_t* inds = (name_t*)malloc(sizeof(name_t)*n);
308 random_numbers(inds, N, n);
309 qsort(inds, n, sizeof(name_t), compare_int);
310 i = 0;
311 for (name_t d = 0; d < ndeme; d++) {
312 while (inds[i] < deme_size[d]) {
313 for (name_t s = 0; s < nseg; s++) {
314 a = inventory[s].get_ball_idx(inds[i],d);
315 geneal[s].sample(a,time());
316 }
317 i++;
318 }
319 }
320 valid_invens(e);
321 free(inds);
322 }
