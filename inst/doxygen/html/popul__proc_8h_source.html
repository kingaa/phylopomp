143 void update_clocks (void) {
144 double rate[nevent];
145 double total_rate = event_rates(rate,nevent);
146 if (total_rate > 0) {
147 next = current+rexp(1/total_rate);
148 } else {
149 next = R_PosInf;
150 }
151 double u = runif(0,total_rate);
152 event = 0;
153 while (u > rate[event] && event < nevent) {
154 if (rate[event] < 0)
155 err("in '%s': invalid rate[%ld]=%lg",__func__,event,rate[event]); // #nocov
156 u -= rate[event];
157 event++;
158 }
159 if (event >= nevent)
160 err("in '%s': invalid event %ld!",__func__,event); // #nocov
161 };

165 int play (double tfin) {
166 int count = 0;
167 if (current > tfin)
168 err("cannot simulate backward! (current t=%lg, requested t=%lg)",current,tfin);
169 while (next < tfin) {
170 current = next;
171 jump(event);
172 update_clocks();
173 count++;
174 R_CheckUserInterrupt();
175 }
176 if (next > tfin) current = tfin; // relies on Markov property
177 return count;
178 };
