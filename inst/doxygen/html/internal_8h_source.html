69inline void random_numbers (name_t* out, size_t len, size_t n) {
70 if (n > len) err("in '%s': exceeding size!",__func__);
71 name_t i, arr[len];
72 for (i = 0; i < len; i++) arr[i] = i;
73 srand(time(NULL));
74 for (i = len - 1; i > 0; i--) {
75 // Pick a random index from 0 to i
76 name_t j = rand() % (i+1);
77 // Swap arr[i] with the element at random index
78 swap_elem(&arr[i], &arr[j]);
79 }
80 for (i = 0; i < n; i++) {
